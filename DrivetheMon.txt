      /* ----- Update Info Box ----- */
      function updateInfoBox() {
        const routeInfoEl = document.getElementById("route-info");
        const distanceInfoEl = document.getElementById("distance-info");
        const timeInfoEl = document.getElementById("time-info");
        const townInfoEl = document.getElementById("town-info");
        const reservoirInfoEl = document.getElementById("reservoir-info");
        if (selectedRoutes.length === 0) {
          routeInfoEl.textContent = "";
          distanceInfoEl.textContent = "";
          timeInfoEl.textContent = "";
        } else if (selectedRoutes.length === 1) {
          const data = routeData[selectedRoutes[0]];
          routeInfoEl.textContent = data.name;
          distanceInfoEl.textContent = data.distance + " miles";
          if (data.time > 60) {
            let hrs = data.time / 60;
            let rounded = Math.ceil(hrs * 4) / 4;
            timeInfoEl.textContent = rounded + " Hours";
          } else {
            timeInfoEl.textContent = data.time + " minutes";
          }
        } else {
          let totalDistance = 0,
            totalTime = 0;
          selectedRoutes.forEach(id => {
            totalDistance += routeData[id].distance;
            totalTime += routeData[id].time;
          });
          routeInfoEl.textContent = "";
          distanceInfoEl.textContent = totalDistance + " miles";
          if (totalTime > 60) {
            let hrs = totalTime / 60;
            let rounded = Math.ceil(hrs * 4) / 4;
            timeInfoEl.textContent = rounded + " Hours";
          } else {
            timeInfoEl.textContent = totalTime + " minutes";
          }
        }
        if (selectedTowns.length === 0) {
          townInfoEl.textContent = "";
          reservoirInfoEl.textContent = "";
        } else {
          const townNames = selectedTowns.map(id => townData[id].name);
          townInfoEl.textContent = townNames.join(" - ");
          const reservoirs = new Set();
          selectedTowns.forEach(id => {
            townData[id].reservoirs.forEach(r => reservoirs.add(r));
          });
          reservoirInfoEl.textContent = Array.from(reservoirs).join(", ");
        }
      }
      /* ----- Toggle Selections ----- */
      function toggleTownSelection(evt) {
        const group = evt.currentTarget;
        const id = group.id;
        if (!townData[id]) return;
        // Get the fill element (assumed to be the path with class "cls-1")
        const fillPath = group.querySelector("path.cls-1");
        if (selectedTowns.includes(id)) {
          // Deselect the town.
          selectedTowns = selectedTowns.filter(item => item !== id);
          if (fillPath) {
            fillPath.style.fill = "#fff";
            // Trigger the pulse animation on deselection (optional)
            fillPath.classList.add("pulse");
            setTimeout(() => {
              fillPath.classList.remove("pulse");
            }, 300);
          }
        } else {
          // Select the town.
          selectedTowns.push(id);
          if (fillPath) {
            fillPath.style.fill = "#B75548";
            // Add the pulse effect upon selection.
            fillPath.classList.add("pulse");
            setTimeout(() => {
              fillPath.classList.remove("pulse");
            }, 300);
          }
        }
        updateInfoBox();
      }

      function toggleRouteSelection(evt) {
        const elem = evt.target;
        const id = elem.id;
        if (!routeData[id]) return;
        if (selectedRoutes.includes(id)) {
          selectedRoutes = selectedRoutes.filter(item => item !== id);
          elem.style.stroke = "#000";
        } else {
          selectedRoutes.push(id);
          elem.style.stroke = "#99CA3C";
        }
        // Add pulse animation class
        elem.classList.add("pulse");
        // Remove the class after the animation duration (300ms)
        setTimeout(() => {
          elem.classList.remove("pulse");
        }, 300);
        updateInfoBox();
      }

      function toggleTownSelection(evt) {
        const group = evt.currentTarget;
        const id = group.id;

        // only change the fill of the element with class "cls-1"

          selectedTowns = selectedTowns.filter(item => item !== id);
          if (fillPath) fillPath.style.fill = "#fff";
        } else {
          selectedTowns.push(id);
          if (fillPath) fillPath.style.fill = "#B75548";
        }

      }

      function toggleTownSelection(evt) {
        const group = evt.currentTarget; // The 
        < g > element
        for the town
        const id = group.id;
        if (!townData[id]) return;
        // Get all 
        < path > elements inside the group.

            p.classList.add("pulse");
            setTimeout(() => {
              p.classList.remove("pulse");
            }, 300);
          });
        } else {
          // Select the town.
          selectedTowns.push(id);
          allPaths.forEach(p => {
            // If it's the fill element, update its fill color.
            if (p.classList.contains("cls-1")) {
              p.style.fill = "#B75548";
            }
            // Add the pulse effect on selection.

      // Function to construct and open a Google Maps loop route URL based on both selected towns and routes.
      function openGoogleMapsRoute() {
        // Combine the selected items from both towns and routes.
        const combinedSelected = [...selectedTowns, ...selectedRoutes];
        // Need at least 2 stops to build a route.
        if (combinedSelected.length < 2) {
          alert("Please select at least two items (towns or routes) to create a driving route.");
          return;
        }
        // Convert each selected item to its GPS coordinate in "lat,lng" format.
        const coords = combinedSelected.map(id => {
          if (townData[id]) {
            const gps = townData[id].gps;
            return gps.lat + "," + gps.lng;
          } else if (routeData[id] && routeData[id].gps) {
            const gps = routeData[id].gps;
            return gps.lat + "," + gps.lng;
          } else {
            return null; // Skip if no GPS data is available.
          }
        }).filter(coord => coord !== null);
        if (coords.length < 2) {
          alert("Not enough GPS coordinates available to build a route.");
          return;
        }
        // Set the first coordinate as both origin and destination to form a loop.
        const origin = coords[0];
        const destination = coords[0];
        // Any remaining coordinates become waypoints.
        let waypoints = "";
        if (coords.length > 1) {
          waypoints = "&waypoints=" + coords.slice(1).join("|");
        }
        // Construct the Google Maps URL.
        const mapUrl = "https://www.google.com/maps/dir/?api=1" + "&origin=" + origin + "&destination=" + destination + waypoints + "&travelmode=driving";
        window.open(mapUrl, "_blank");
      }
